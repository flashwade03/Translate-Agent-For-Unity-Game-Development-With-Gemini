```python
def delete_rows(self, project_id: str, sheet_name: str, keys: list[str]) -> int:
    """Delete rows by key names. Returns count of actually deleted rows."""
    csv_path = self._csv_path(project_id, sheet_name)
    if not csv_path.exists():
        return 0
    keys_set = set(keys)
    with open(csv_path, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        headers = reader.fieldnames or []
        rows = list(reader)
    original_count = len(rows)
    rows = [r for r in rows if r["key"] not in keys_set]
    with open(csv_path, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=headers)
        writer.writeheader()
        writer.writerows(rows)
    return original_count - len(rows)
```

```python
    active_jobs = [
        j for j in job_svc.jobs.values()
        if j.project_id == project_id
        and j.sheet_name == sheet_name
        and j.status in ("pending", "running")
    ]
    if active_jobs:
        raise HTTPException(409, "Cannot delete rows with active translation jobs")
```

```typescript
export function deleteRows(projectId: string, sheetName: string, keys: string[]) {
  return api<{ ok: boolean; deletedCount: number }>(
    'DELETE',
    `/api/projects/${projectId}/sheets/${encodeURIComponent(sheetName)}/rows`,
    { keys },
  )
}
```

```typescript
  if (data) {
```
